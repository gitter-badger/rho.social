 58     /**

 64      * @param User $identity
 65      * @return static
 66      * @throws NotFoundHttpException
 67      */
 68     public static function findOneByIdentity($id, $throwException = true, $identity = null)
 69     {
 70         $query = static::findByIdentity($identity);
 71         if (!empty($id)) {
 72             $query = $query->id($id);
 73         }
 74         $model = $query->one();
 75         if (!$model && $throwException) {
